  • forEach(slice:body:)

    将指定 Collection 依照给定 slice 大小分割，并在给定函数闭包中返回分割结果

    var iterations = 0
let array = ["a", "b", "c", "d", "e", "f", "g"]
array.forEach(slice: 2, body: { (slices) in
    switch iterations {
    case 0:
        slices -> ["a", "b"]
    case 1:
        slices -> ["c", "d"]
    case 2:
        slices -> ["e", "f"]
    case 3:
        slices -> ["g"]
    default:
        break
    }
    iterations += 1
})
